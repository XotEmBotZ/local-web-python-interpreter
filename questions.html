        total = 0
        s = input("Enter a number or 'done': ")
        while s != 'done':
            s = int(s)

            s = input("Enter a number or 'done': ")
        print("Total sum of numbers is", total)
    sumnum()

    def name_fun(s):

    count = 0

Write a Python program to input two numbers and calculate and print their LCM (least common multiple). Example: If the numbers are 9 and 12, then LCM is 36.

Author:TANISH KUMAR

Show Answer: 
    import math
    def lcm(a, b):
        lcm = (a * b) // math.gcd(a, b)
        return lcm
    print(lcm(9, 12))

Show Answer: 
    def countk():
        f = open("hello.txt", "r")
        C = 0
        l = f.readlines()
        for i in l:
            if i[0].lower() == 'k':
                C += 1
        print("No of lines are:", C)
        f.close()
